
        fibonacci: function integer(n: integer) {
            f0,f1,fn: integer = 0,1,1;
            i:integer;
            if (n < 0) {
                return -1;
            } 
            if ((n == 0) || (n == 1)) {
                return n;
            } else {
                for (i = 2, i < n, i+1) {
                    f0 = f1;
                    f1 = fn;
                    fn = f0 + f1;
                }
            }
            return fn;
        }
        simpleFunc: function void () {
            print(fibonacci(10));
        }
